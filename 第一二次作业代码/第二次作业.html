remove_outlier = function(col_name, data){
  for(cn in col_name){
    # 洗离群值
    Q1 <- quantile(data[,cn], 0.25)
    Q3 <- quantile(data[,cn], 0.75)
    IQR <- Q3 - Q1
    lower_bound <- Q1 - 1.5 * IQR
    upper_bound <- Q3 + 1.5 * IQR
    data <- data[data[,cn]>=lower_bound & data[,cn]<=upper_bound, ]
  }
  return(data)
}
